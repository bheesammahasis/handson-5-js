  Sample Input

    const countCharacters = (S) => 
    {
         let ansA=0;
         let ansD=0;
         for(i=0;i< S.length;i++){
         if(S.charAt(i) === 'A'){
           ansA++;
         }
         else if(S.charAt(i)==='D'){
            ansD++;
         }
        }
          let ans=[ansA,ansD];
          return ans;
    };

    Example
    Sample Input 1
    Naman
    Sample Output 1
    False
    Sample Input 2
    naman
    Sample Output 2
    True
 
    ANSWER NO8:-
    var Palin_Check = (str) =>
    {
	  const n = str.length;
     // loop through half of the string
      for (let i = 0; i < n / 2; i++) {
     // check if first and last string are same
      if (str[i] !== str[n - 1 - i]) {
            return 'False';
        }
     }
       return 'True';
    }
